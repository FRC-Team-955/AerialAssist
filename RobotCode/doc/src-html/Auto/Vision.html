001    package Auto;
002    
003    import Utils.Config;
004    import edu.wpi.first.wpilibj.networktables.NetworkTable;
005    
006    /**
007     *
008     * @author RaiderPC
009     */
010    public class Vision 
011    {
012        private NetworkTable table;
013    
014        public Vision()
015        {
016            table = NetworkTable.getTable(Config.Vision.tableId);
017            resetTable();
018        }
019        
020        /**
021         * Check if raspberry pi found hot target
022         * @return 
023         */
024        public boolean foundHotTarget()
025        {
026            return table.getBoolean(Config.Vision.isGoalHotId);
027        }
028        
029        /**
030         * Tell raspberry pi side preference for hot target
031         * @param prefLeft 
032         */
033        public void setPrefSideLeft(boolean prefLeft)
034        {
035            table.putBoolean(Config.Vision.prefSideLeftId, prefLeft);
036        }
037        
038        /**
039         * Turn off the raspberry pi
040         */
041        public void turnOffPi()
042        {
043            table.putBoolean(Config.Vision.shutDownId, true);
044        }
045        
046        /**
047         * Set debug mode for raspberry pi
048         * @param state 
049         */
050        public void setDebugMode(boolean state)
051        {
052            table.putBoolean(Config.Vision.debugModeId, state);
053        }
054        
055        /**
056         * Initializes all the values in the vision network table
057         */
058        public void resetTable()
059        {
060            table.putBoolean(Config.Vision.prefSideLeftId, false);
061            table.putBoolean(Config.Vision.debugModeId, true);
062            table.putBoolean(Config.Vision.shutDownId, false);
063            table.putBoolean(Config.Vision.isGoalHotId, false);
064        }
065    }
