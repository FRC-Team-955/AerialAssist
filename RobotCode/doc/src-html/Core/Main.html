008    package Core;
009    
010    import ModClasses.MyJoystick;
011    import ModClasses.Station;
012    import Utils.Config;
013    import Auto.Autonomous;
014    import edu.wpi.first.wpilibj.IterativeRobot;
015    import edu.wpi.first.wpilibj.Compressor;
016    import edu.wpi.first.wpilibj.Talon;
017    import edu.wpi.first.wpilibj.Timer;
018    /**
019     * The VM is configured to automatically run this class, and to call the
020     * functions corresponding to each mode, as described in the IterativeRobot
021     * documentation. If you change the name of this class or the package after
022     * creating this project, you must also update the manifest file in the resource
023     * directory.
024     */
025    public class Main extends IterativeRobot 
026    {
027            String codeVersion = "CODE FOR OSU DISTRICT";
028        MyJoystick joy = new MyJoystick(Config.Joystick.chn);
029        Drive drive = new Drive(joy);
030        Catapult catapult = new Catapult(joy);
031        Pickup pickUp = new Pickup(joy);
032        Autonomous auto = new Autonomous(catapult, drive, pickUp);
033        Compressor compressor = new Compressor(Config.Compressor.chnDigInPressure, Config.Compressor.chnDigOutCompressor);
034            Test test = new Test(drive, catapult, pickUp);
035        
036        /**
037         * This function is run when the robot is first started up and should be
038         * used for any initialization code.
039         */
040        public void robotInit()
041        {
042            // Setting to get regular drive working properly on the ps3 Controller should be 3, 2
043            joy.setAxisChannel(MyJoystick.AxisType.kX, 3);
044            joy.setAxisChannel(MyJoystick.AxisType.kY, 2);
045            pickUp.up();
046            System.out.println("Init called: " + codeVersion);
047        }
048        
049        /**
050         * This function is called once before autonomous
051         */
052        public void autonomousInit()
053        {
054            auto.init();
055        }
056        
057        /**
058         * This function is called periodically during autonomous
059         */
060        public void autonomousPeriodic()
061        {
062            Station.clearAllText();
063            auto.run();
064        }
065        
066        /**
067         * This function is called once before teloepPeriodic
068         */
069        public void teleopInit()
070        {
071            compressor.start();
072            auto.end();
073        }
074        
075        /**
076         * This function is called periodically during operator control
077         */
078        public void teleopPeriodic() 
079        {
080            Station.clearAllText();
081            Station.print(Config.Station.mainCodeType, codeVersion);
082            joy.updateButtons();
083            drive.run();
084            catapult.run();
085            pickUp.run();
086        }
087    
088        /**
089         * This function is called once before robot is disabled.
090         */
091        public void disabledInit()
092        {
093            auto.resetVision();
094        }
095            
096            /**
097             * This function is called once before testPeriodic
098             */
099            public void testInit()
100            {
101                    test.reset();
102            }
103            
104            /**
105             * This function is called periodically during test
106             */
107            public void testPeriodic()
108            {
109                    test.run();
110            }
111    }
