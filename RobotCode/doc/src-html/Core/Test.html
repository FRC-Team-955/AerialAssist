001    package Core;
002    
003    import edu.wpi.first.wpilibj.Timer;
004    
005    /**
006     *
007     * @author raiderbot-4
008     */
009    public class Test 
010    {
011            private int i = 1;
012            private Timer testTimer = new Timer();
013            private boolean running = true;
014            private boolean starting = true;
015            private Drive drive;
016            private Catapult cat;
017            private Pickup pickup;
018            
019            public Test(Drive drive, Catapult cat, Pickup pickup)
020            {
021                    this.drive = drive;
022                    this.cat = cat;
023                    this.pickup =  pickup;
024            }
025                            
026            /**
027             * Runs the test routine
028             */
029            public void run()
030            {
031                    if(starting)
032                    {
033                            testTimer.start();
034                            starting = false;
035                    }
036                    
037                    if(testTimer.get() > .5 && running)
038                    {
039                            switch(i) 
040                            {
041                                    case 1:
042                                            testTimer.reset();
043                                            drive.setMotorLeft1(1);
044                                            break;
045                                    case 2: 
046                                            testTimer.reset();
047                                            drive.setMotorLeft1(0);
048                                            drive.setMotorLeft2(1);
049                                            break;
050                                    case 3: 
051                                            testTimer.reset();
052                                            drive.setMotorLeft2(0);
053                                            drive.setMotorLeft3(1);
054                                            break;
055                                    case 4: 
056                                            testTimer.reset();
057                                            drive.setMotorLeft3(0);
058                                            drive.setMotorRight1(1);
059                                            break;
060                                    case 5: 
061                                            testTimer.reset();
062                                            drive.setMotorRight1(0);
063                                            drive.setMotorRight2(1);
064                                            break;
065                                    case 6: 
066                                            testTimer.reset();
067                                            drive.setMotorRight2(0);
068                                            drive.setMotorRight3(1);
069                                            break;
070                                    case 7: 
071                                            testTimer.reset();
072                                            drive.setMotorRight3(0);
073                                            cat.testCatapult(1);
074                                            break;
075                                    case 8: 
076                                            testTimer.reset();
077                                            cat.testCatapult(0);
078                                            pickup.testPickup(1);
079                                            break;
080                                    case 9: 
081                                            testTimer.reset();
082                                            pickup.testPickup(0);
083                                            running = false;
084                                            i = 1;
085                                            break;
086                            }
087                            i+=1;
088                    }
089            }
090            
091            /**
092             * Resets everything back to original 
093             */
094            public void reset()
095            {
096                    starting = true;
097                    running = true;
098                    i = 1;
099            }
100    }
