001    package ModClasses;
002    
003    import edu.wpi.first.wpilibj.Solenoid;
004    
005    /**
006     *
007     * @author coders
008     */
009    public class MySolenoid 
010    {
011        //Declares solenoids
012        private Solenoid solOne;
013        private Solenoid solTwo;
014        
015        /**
016         * Creates the solenoids
017         * @param portOne   the slot the solenoid bumper is plugged into the cRio
018         * @param portTwo   the ports on the solenoid bumper
019         */
020        public MySolenoid(int portOne, int portTwo) 
021        {
022            //Creates solenoids
023            solOne = new Solenoid(portOne);
024            solTwo = new Solenoid(portTwo);      
025        }
026        
027        /**
028         * Switches the solenoids between on and off positions
029         */
030        public void flip()
031        {
032            //Switches the solenoids, one is true, one false, just switches which in is which
033            if (solOne.get()) 
034            {
035                solOne.set(false);
036                solTwo.set(true);
037            }
038            
039            //Switches in the other instance, when solenoidTwo is true, not solenoidOne
040            else 
041            {
042                solOne.set(true);
043                solTwo.set(false);
044            }
045        }   
046        
047        /**
048         * Sets the solenoids to a specific state
049         * @param newState 
050         */
051        public void set(boolean newState)
052        {
053            if(newState)
054            {
055                solOne.set(true);
056                solTwo.set(false);
057            }
058            
059            else
060            {
061                solOne.set(false);
062                solTwo.set(true);
063            }
064        }
065        
066        /**
067         * Returns the position of the solenoid
068         * @return 
069         */
070        public boolean get()
071        {
072            return solOne.get();
073        }
074    }
