001    package ModClasses;
002    
003    import edu.wpi.first.wpilibj.Talon;
004    import Utils.Config;
005    
006    /**
007     *
008     *  @author raiderbot-4
009     */
010    public class MyTalon extends Talon 
011    {
012        public MyTalon(int chan)
013        {
014            super(chan);
015        }
016        
017        /**
018         * Makes sure the motors do not accelerate too fast
019         * @param newSpeed The speed that was requested
020         */
021        public void ramp(double newSpeed)
022        {
023            double curSpeed = super.get();
024            double difference = Math.abs(newSpeed-curSpeed);
025            
026            if(difference > Config.MyTalon.rampRate)
027            {
028                if(newSpeed > curSpeed)
029                    newSpeed = curSpeed + Config.MyTalon.rampRate;
030                
031                else
032                    newSpeed = curSpeed - Config.MyTalon.rampRate;
033            } 
034             
035            set(newSpeed);      
036        }
037    }
