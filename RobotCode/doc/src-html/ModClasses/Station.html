001    package ModClasses;
002    
003    import edu.wpi.first.wpilibj.DriverStation;
004    import edu.wpi.first.wpilibj.DriverStationLCD;
005    
006    /**
007     *
008     * @author Merfoo
009     */
010    public class Station 
011    {
012        /**
013         * Gets the button status from the driverstation, 1 - 8 available.
014         * @param iChan
015         * @return 
016         */
017        public static boolean getDigitalIn(int iChan)
018        {
019            return DriverStation.getInstance().getDigitalIn(iChan);
020        }
021        
022        /**
023         * Gets the analog value from the driverstation, 1 - 8 available.
024         * @param iChan
025         * @return 
026         */
027        public static double getAnalogIn(int iChan)
028        {
029            return DriverStation.getInstance().getAnalogIn(iChan);
030        }
031    
032        /**
033         * Prints specified message to the driver station on the corresponding line
034         * 1-6 are available.
035         * @param line
036         * @param msg
037         */
038        public static void print(int line, String msg)
039        {
040            DriverStationLCD.Line printLine = null;
041            
042            switch(line)
043            {
044                case 1: printLine = DriverStationLCD.Line.kUser1; break;
045                case 2: printLine = DriverStationLCD.Line.kUser2; break;
046                case 3: printLine = DriverStationLCD.Line.kUser3; break;
047                case 4: printLine = DriverStationLCD.Line.kUser4; break;
048                case 5: printLine = DriverStationLCD.Line.kUser5; break;
049                case 6: printLine = DriverStationLCD.Line.kUser6; break;
050            }
051            
052            DriverStationLCD.getInstance().println(printLine, 1, msg);
053            DriverStationLCD.getInstance().updateLCD(); 
054        }
055        
056        /**
057         * Clears all the text in the driver station
058         */
059        public static void clearAllText()
060        {
061            DriverStationLCD.getInstance().clear();
062        }
063    }
