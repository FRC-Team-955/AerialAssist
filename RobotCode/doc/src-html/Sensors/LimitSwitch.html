001    package Sensors;
002    
003    import edu.wpi.first.wpilibj.DigitalInput;
004    
005    /**
006     *
007     * @author RaiderPC
008     */
009    public class LimitSwitch 
010    {
011        private DigitalInput input;
012        private boolean flipped;
013        
014        public LimitSwitch(int port, boolean wantFlipped)
015        {
016                input = new DigitalInput(port);
017                flipped = wantFlipped;
018        }
019    
020        /**
021         * Returns position of the limit switch
022         * @return 
023         */
024        public boolean get() 
025        {
026            boolean value = input.get();
027    
028            if (flipped)
029                    value = !value;
030    
031            return value;
032        }
033    }
